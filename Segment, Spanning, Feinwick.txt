SegmentTree
-----------------------------------------
#include <bits/stdc++.h>
using namespace std;

class SegmentTree {
    vector<int> tree;
    int n;
    
public:
    // Constructor
    SegmentTree(vector<int>& arr) {
        n = arr.size();
        tree.resize(4 * n);  // 4*n is safe size
        build(arr, 1, 0, n-1);
    }
    
    // Build tree from array
    void build(vector<int>& arr, int node, int start, int end) {
        if(start == end) {
            tree[node] = arr[start];  // leaf node
        } else {
            int mid = (start + end) / 2;
            build(arr, 2*node, start, mid);      // left child
            build(arr, 2*node+1, mid+1, end);    // right child
            tree[node] = tree[2*node] + tree[2*node+1];  // sum
        }
    }
    
    // Query sum in range [L,R]
    int query(int L, int R) {
        return queryUtil(1, 0, n-1, L, R);
    }
    
    int queryUtil(int node, int start, int end, int L, int R) {
        // No overlap
        if(R < start || end < L) return 0;
        
        // Total overlap
        if(L <= start && end <= R) return tree[node];
        
        // Partial overlap
        int mid = (start + end) / 2;
        int leftSum = queryUtil(2*node, start, mid, L, R);
        int rightSum = queryUtil(2*node+1, mid+1, end, L, R);
        return leftSum + rightSum;
    }
    
    // Update value at index idx to newVal
    void update(int idx, int newVal) {
        updateUtil(1, 0, n-1, idx, newVal);
    }
    
    void updateUtil(int node, int start, int end, int idx, int newVal) {
        if(start == end) {
            tree[node] = newVal;  // leaf node
        } else {
            int mid = (start + end) / 2;
            if(idx <= mid) {
                updateUtil(2*node, start, mid, idx, newVal);  // left
            } else {
                updateUtil(2*node+1, mid+1, end, idx, newVal); // right
            }
            tree[node] = tree[2*node] + tree[2*node+1];  // update parent
        }
    }
};

int main() {
    vector<int> arr = {1, 3, 5, 7, 9, 11};
    
    SegmentTree st(arr);
    
    cout << "Sum [1,3]: " << st.query(1, 3) << endl;  // 3+5+7 = 15
    cout << "Sum [0,5]: " << st.query(0, 5) << endl;  // 36
    
    st.update(2, 10);  // change 5 to 10
    
    cout << "Sum [1,3] after update: " << st.query(1, 3) << endl;  // 3+10+7 = 20
    
    return 0;
}

Find Any Spanning Tree: DFS
------------------------------------
#include <bits/stdc++.h>
using namespace std;

void dfs(int node, vector<vector<int>>& adj, vector<bool>& vis, 
         vector<pair<int,int>>& spanningTree) {
    vis[node] = true;
    
    for(int neighbor : adj[node]) {
        if(!vis[neighbor]) {
            // Add edge to spanning tree
            spanningTree.push_back({node, neighbor});
            dfs(neighbor, adj, vis, spanningTree);
        }
    }
}

int main() {
    int n = 5; // vertices
    vector<vector<int>> adj(n);
    
    // Add edges (undirected)
    adj[0].push_back(1); adj[1].push_back(0);
    adj[0].push_back(3); adj[3].push_back(0);
    adj[1].push_back(2); adj[2].push_back(1);
    adj[2].push_back(3); adj[3].push_back(2);
    adj[2].push_back(4); adj[4].push_back(2);
    adj[3].push_back(4); adj[4].push_back(3);
    
    vector<bool> vis(n, false);
    vector<pair<int,int>> spanningTree;
    
    dfs(0, adj, vis, spanningTree);
    
    cout << "Spanning Tree edges:" << endl;
    for(auto& [u,v] : spanningTree) {
        cout << u << " - " << v << endl;
    }
    // Output: 4 edges for 5 vertices
    
    return 0;
}

Fenwick Tree
----------------------------------
#include <bits/stdc++.h>
using namespace std;

class Fenwick {
    vector<int> bit;
    int n;

public:
    Fenwick(int n) {
        this->n = n;
        bit.assign(n + 1, 0);
    }

    void update(int i, int val) {
        while (i <= n) {
            bit[i] += val;
            i += (i & -i);
        }
    }

    int sum(int i) {
        int s = 0;
        while (i > 0) {
            s += bit[i];
            i -= (i & -i);
        }
        return s;
    }

    int rangeSum(int l, int r) {
        return sum(r) - sum(l - 1);
    }
};
